function moverGuerrero() {
  var guerrero = document.getElementById("Zeus");
  var posicionActualTop = parseInt(guerrero.style.top);
  var posicionActualLeft = parseInt(guerrero.style.left);
  
  var nuevaPosicionTop = posicionActualTop + 10; // Mover hacia abajo
  var nuevaPosicionLeft = posicionActualLeft + 10; // Mover hacia la derecha
  
  guerrero.style.top = nuevaPosicionTop + "10px";
  guerrero.style.left = nuevaPosicionLeft + "10px";
}
